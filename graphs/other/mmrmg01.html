MMRMG01

Plots for Mixed-Effect Model Repeated Measures Analysis

Given an MMRM fitted with s_mmrm, g_mmrm_lsmeans displays for each visit the adjusted means within group and/or difference in adjusted means between groups. g_mmrm_diagnostic displays marginal residual plots for evaluating model fit.

library(dplyr)
library(tern.mmrm)
library(scda)
library(nestcolor)

adsl <- synthetic_cdisc_dataset("latest", "adsl")
adqs <- synthetic_cdisc_dataset("latest", "adqs")

adqs_f <- adqs %>%
  dplyr::filter(PARAMCD == "FKSI-FWB" & !AVISIT %in% c("BASELINE")) %>%
  droplevels() %>%
  dplyr::mutate(ARM = factor(ARM, levels = c("B: Placebo", "A: Drug X", "C: Combination"))) %>%
  dplyr::mutate(AVISITN = rank(AVISITN) %>% as.factor() %>% as.numeric() %>% as.factor())

mmrm_results <- fit_mmrm(
  vars = list(
    response = "AVAL",
    covariates = c("STRATA2"),
    id = "USUBJID",
    arm = "ARM",
    visit = "AVISIT"
  ),
  data = adqs_f
)

Considering the treatment variable in the model

Code
g_mmrm_lsmeans(
  mmrm_results,
  select = "estimates",
  xlab = "Visit"
)

g_mmrm_lsmeans(

Code
g_mmrm_lsmeans(

library(teal.modules.clinical)
library(scda)

adsl <- synthetic_cdisc_dataset("latest", "adsl")
adqs <- synthetic_cdisc_dataset("latest", "adqs") %>%
  dplyr::filter(ABLFL != "Y" & ABLFL2 != "Y") %>%
  dplyr::filter(AVISIT %in% c("WEEK 1 DAY 8", "WEEK 2 DAY 15", "WEEK 3 DAY 22")) %>%
  dplyr::mutate(
    AVISIT = as.factor(AVISIT),
    AVISITN = rank(AVISITN) %>%
      as.factor() %>%
      as.numeric() %>%
      as.factor() # making consecutive numeric factor
  )

arm_ref_comp <- list(
  ARMCD = list(
    ref = "ARM A",
    comp = c("ARM B", "ARM C")
  )
)

app <- init(
  data = cdisc_data(
    cdisc_dataset("ADSL", adsl),
    cdisc_dataset("ADQS", adqs),
    code = '
    ADSL <- synthetic_cdisc_dataset("latest", "adsl")
    ADQS <- synthetic_cdisc_dataset("latest", "adqs") %>%
              dplyr::filter(ABLFL != "Y" & ABLFL2 != "Y") %>%
              dplyr::filter(AVISIT %in% c("WEEK 1 DAY 8", "WEEK 2 DAY 15", "WEEK 3 DAY 22")) %>%
              dplyr::mutate(
                AVISIT = as.factor(AVISIT),
                AVISITN = rank(AVISITN) %>%
                  as.factor() %>%
                  as.numeric() %>%
                  as.factor() # making consecutive numeric factor
              )
    ',
    check = TRUE
  ),
  modules = modules(
    tm_a_mmrm(
      label = "MMRM",
      dataname = "ADQS",
      aval_var = choices_selected(c("AVAL", "CHG"), "AVAL"),
      id_var = choices_selected(c("USUBJID", "SUBJID"), "USUBJID"),
      arm_var = choices_selected(c("ARM", "ARMCD"), "ARMCD"),
      visit_var = choices_selected(c("AVISIT", "AVISITN"), "AVISIT"),
      arm_ref_comp = arm_ref_comp,
      paramcd = choices_selected(
        choices = value_choices(adqs, "PARAMCD", "PARAM"),
        selected = "FKSI-FWB"
      ),
      cov_var = choices_selected(c("BASE", "AGE", "SEX", "BASE:AVISIT"), NULL),
      conf_level = choices_selected(c(0.95, 0.9, 0.8), 0.95)
    )
  )
)

Given an MMRM fitted with `s_mmrm`, `g_mmrm_lsmeans` displays for each visit the adjusted means within group and/or difference in adjusted means between groups.
`g_mmrm_diagnostic` displays marginal residual plots for evaluating model fit.

::: panel-tabset
## Data Setup and <br/> Model Fitting

```{r setup, message=FALSE}
#| code-fold: show

library(dplyr)
library(tern.mmrm)
library(scda)
library(nestcolor)

adsl <- synthetic_cdisc_dataset("latest", "adsl")
adqs <- synthetic_cdisc_dataset("latest", "adqs")

adqs_f <- adqs %>%
  dplyr::filter(PARAMCD == "FKSI-FWB" & !AVISIT %in% c("BASELINE")) %>%
  droplevels() %>%
  dplyr::mutate(ARM = factor(ARM, levels = c("B: Placebo", "A: Drug X", "C: Combination"))) %>%
  dplyr::mutate(AVISITN = rank(AVISITN) %>% as.factor() %>% as.numeric() %>% as.factor())

mmrm_results <- fit_mmrm(
  vars = list(
    response = "AVAL",
    covariates = c("STRATA2"),
    id = "USUBJID",
    arm = "ARM",
    visit = "AVISIT"
  ),
  data = adqs_f
)
```

## Least Squares Means: <br/> Estimates Within Groups

### Considering the treatment variable in the model

```{r, dev.args=list(pointsize=6)}
g_mmrm_lsmeans(
  mmrm_results,
  select = "estimates",
  xlab = "Visit"
)
```

### Considering the treatment variable in the model, with lines

```{r, dev.args=list(pointsize=6)}
g_mmrm_lsmeans(
  mmrm_results,
  select = "estimates",
  xlab = "Visit",
  show_lines = TRUE
)
```

### Considering the treatment variable in the model, with statistics table

```{r, dev.args=list(pointsize=6), fig.height=7}
g_mmrm_lsmeans(
  mmrm_results,
  select = "estimates",
  xlab = "Visit",
  table_stats = c("n", "estimate", "se", "ci"),
  table_font_size = 4,
  table_rel_height = 0.6
)
```

## Least Squares Means: <br/> Contrasts Between Groups

Users can choose to display both estimates and contrasts together by running `g_mmrm_lsmeans(mmrm_results)`.

```{r, dev.args=list(pointsize=6)}
g_mmrm_lsmeans(
  mmrm_results,
  select = "contrasts",
  titles = c(contrasts = "Contrasts of FKSI-FWB means"),
  xlab = "Visit"
)
```

## Model Diagnostics: Marginal <br/> Fitted Values vs. Residuals

```{r, dev.args=list(pointsize=6)}
g_mmrm_diagnostic(mmrm_results)
```

## Model Diagnostics: Normality <br/> of Marginal Residuals

```{r, dev.args=list(pointsize=6)}
g_mmrm_diagnostic(mmrm_results, type = "q-q-residual")
```
