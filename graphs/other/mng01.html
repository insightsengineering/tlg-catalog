library(scda)
library(dplyr)
library(teal.modules.clinical)

ADSL <- synthetic_cdisc_dataset("latest", "adsl")
ADLB <- synthetic_cdisc_dataset("latest", "adlb") %>%
  mutate(AVISIT = forcats::fct_reorder(AVISIT, AVISITN, min))

app <- init(
  data = cdisc_data(
    cdisc_dataset("ADSL", ADSL, code = 'ADSL <- synthetic_cdisc_dataset("latest", "adsl")'),
    cdisc_dataset("ADLB", ADLB, code = 'ADLB <- synthetic_cdisc_dataset("latest", "adlb") %>%
                    mutate(AVISIT = forcats::fct_reorder(AVISIT, AVISITN, min))'),
    check = TRUE
  ),
  modules = modules(
    tm_g_lineplot(
      label = "LINE PLOT",
      dataname = "ADLB",
      strata = choices_selected(
        variable_choices(ADSL, c("ARM", "ARMCD", "ACTARMCD")),
        "ARM"
      ),
      y = choices_selected(
        variable_choices(ADLB, c("AVAL", "BASE", "CHG", "PCHG")),
        "AVAL"
      ),
      param = choices_selected(
        value_choices(ADLB, "PARAMCD", "PARAM"),
        "ALT"
      )
    )
  )
)
