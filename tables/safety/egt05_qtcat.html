adsl <- synthetic_cdisc_dataset("rcd_2022_10_13", "adsl")
adeg <- synthetic_cdisc_dataset("rcd_2022_10_13", "adeg")

# Ensure character variables are converted to factors and empty strings and NAs are explicit missing levels.
adsl <- df_explicit_na(adsl)
adeg <- df_explicit_na(adeg)
adeg_labels <- var_labels(adeg)
adeg_f <- adeg %>%
  filter(
    PARAMCD == "QT",
    ANL01FL == "Y"
  ) %>%
  mutate(
    AVALCAT1 = case_when(
      AVAL <= 450 ~ "<=450 msec",
      AVAL <= 480 ~ ">450 to <=480 msec",
      AVAL <= 500 ~ ">480 to <= 500 msec",
      AVAL > 500 ~ ">500 msec",
      is.na(AVAL) ~ "<Missing>"
    ),
    CHGCAT1 = case_when(
      CHG <= 30 ~ "<=30 msec",
      CHG <= 60 ~ ">30 to <=60 msec",
      CHG > 60 ~ ">60 msec",
      is.na(CHG) ~ "<Missing>"
    )
  ) %>%
  mutate(
    AVALCAT1 = factor(
      AVALCAT1,
      levels = c(
        "<=450 msec",
        ">450 to <=480 msec",
        ">480 to <= 500 msec",
        ">500 msec",
        "<Missing>"
      )
    ),
    CHGCAT1 = factor(
      CHGCAT1,
      levels = c(
        "<=30 msec",
        ">30 to <=60 msec",
        ">60 msec",
        "<Missing>"
      )
    )
  )
var_labels(adeg_f) <- c(
  adeg_labels,
  "AVALCAT1" = "Value at Visit",
  "CHGCAT1" = "Change from Baseline"
)
